      "Hola soy Chato, ven platiquemos"]

#mensaje de vienvenida del chatbot
print("Hola soy chato, tu Chatbot de confianza, si deseas terminar escribe 'salir'")

while True:
    usuario=input("tu:").lower().strip()

    if usuario in saludos:
        print("Chatbot:",random.choice(respuesta_saludos))

    elif usuario in preguntas:
        print("Chatbot:",random.choice(respuestas_preguntas))

    elif usuario in pregunt:
        print("Chatbot:",random.choice(resp))

    elif usuario in despedida:
        print("Chatbot:",random.choice(respuesta_despedida))

        break
